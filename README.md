# Vokalzähler | Version 1.0 Alpha - www.github.com/TimsLab

data = input('Gib einen Satz ein: ')
data = data.lower()
vokale = ['a','e','i','o','u']
vokal = 0

for string in data:
    for i in vokale:
        if string == i:
            vokal += 1

print('Der Satz hat %s Vokale.' %(vokal))
print('Made by www.github.com/TimsLab')
